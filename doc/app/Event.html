def as_json(options = {})
  {
      :id => self.id,
      :title => self.name,
      :description => self.description || "",
      :start => self.date_start,
      :end => self.date_start,
      :allDay => false,
      :recurring => false,
      :url => Rails.application.routes.url_helpers.event_path(id)
  }
end
